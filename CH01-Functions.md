\begin{table}[h!]
\centering
\caption{Types of Functions}
\begin{tabular}{|l|l|p{8cm}|}
\hline
\textbf{Type} & \textbf{Equation} & \textbf{Definition} \\
\hline
Polynomial & \( f(x) = a_n x^n + a_{n-1} x^{n-1} + \dots + a_1 x + a_0 \) & A function that is a sum of terms with non-negative integer powers of \(x\) and constant coefficients. \\
\hline
Exponential & \( f(x) = a \cdot b^x \) & A function in which the variable \(x\) appears in the exponent. \\
\hline
Logarithmic & \( f(x) = \log_b(x) \) & The inverse of an exponential function; gives the power to which the base \(b\) must be raised to get \(x\). \\
\hline
Trigonometric & \( f(x) = \sin x, \cos x, \tan x, \dots \) & Functions based on angles, relating ratios of sides in a right triangle or points on the unit circle. \\
\hline
Inverse & \( f(x) = \frac{1}{x} \) & A function that “reverses” another function, often of the form \( f(x) = x^{-1} \). \\
\hline
Piecewise & \( f(x) = \begin{cases} x^2 & x \ge 0 \\ -x & x < 0 \end{cases} \) & A function defined by different expressions over different intervals of the domain. \\
\hline
\end{tabular}
\end{table}
